```C++
//分解质因数  
//算数基本定理：一个数要么为质数，要么可以被唯一确定的质因数相乘得到  
void dfs(int n, int p = 2) {  
    if (n == 1)  
        return;  
    //如果 n % p == 0 就说明 p 是 n 的一个因数  
    //如果 p 为一个质数，输出  
    //如果 p 为一个合数，那么它会被之前的质数分解掉，因此不需要检测是否为合数  
    //例如 4 被 2 分解掉，9 被 3 分解掉  
    if (n % p == 0){  
        cout << p;  
        dfs(n / p, p);  
    }  
    else  
        dfs(n, p + 1);  
}
```
